module.exports = class InputProcessor
  constructor: ->
    @redis      = new Redis
    @redisBlock = new Redis
    @wait()

  wait: ->
    @redisBlock.blpop "schlep", 0, (err, data) =>
      console.log err if err
      @handleData data[1]
      @wait()

  handleData: (data) ->
    envelope = new Envelope(data)
